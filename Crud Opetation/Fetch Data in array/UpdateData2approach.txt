package jf.university.api;

import java.io.IOException;
import java.io.PrintWriter;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import org.json.JSONObject;

/**
 * Servlet implementation class UniversityUpdateDetails
 */
@WebServlet("/UniversityUpdateDetails")
public class UniversityUpdateDetails extends HttpServlet {
	
	private static final long serialVersionUID = 1L;
	private PrintWriter pw = null;
	private JSONObject jsonObject  = null; 
	
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		// TODO Auto-generated method stub
		pw = response.getWriter();
		
		long   postUid     = Long.parseLong(request.getParameter("getUid"));
		String postUname   = request.getParameter("getUname");
		
		String postCname   = request.getParameter("getCname");
		String postcode    = request.getParameter("getCcode");
		String postmobile  = request.getParameter("getCmobile");
		String postCmobile = postcode+"-"+postmobile;
		
		String postNodept  = request.getParameter("getNodept");
		
		String postAddress = request.getParameter("getAddress");
		String postCity    = request.getParameter("getCity");
		int postPincode    = Integer.parseInt(request.getParameter("getPincode"));
		
		jsonObject = UniversityAPI.updateUniversityDetails(postUid,postUname,postCname,postCmobile,postNodept,postAddress,postCity,postPincode);
		pw.println(jsonObject);
	}

}

public static JSONObject updateUniversityDetails(long pId,String postUname,String postCname,String postCmobile,String postNodept,String postAddress,String postCity,int postPincode)
		{
			JSONObject obj = new JSONObject();
			con = Dao.getConnection();
			String query = "UPDATE university SET uname=?,cname=?, cmobile=?,nodept=?,address=?,city=?,pincode=? WHERE uid=?";
			try
			{
				pst = con.prepareStatement(query);
				pst.setString(1, postUname);
				pst.setString(2, postCname);
				pst.setString(3, postCmobile);
				pst.setString(4, postNodept);
				pst.setString(5, postAddress);
				pst.setString(6, postCity);
				pst.setInt(7, postPincode);
				pst.setLong(8, pId);
				
				int result = pst.executeUpdate();
				if(result == 1)
				{
					obj.put("flag", "Updated successfully.");
				}
				else 
				{
					obj.put("flag", "Issue in Updating Data.");
				}
			}
			catch (SQLException | JSONException e) 
			{
				e.printStackTrace();
			}
			finally {
				Dao.cleanUp(con, pst, null);
			}

			return obj;
		}
